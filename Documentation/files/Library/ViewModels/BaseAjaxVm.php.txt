class BaseAjaxVm extends BaseVm implements \Library\Interfaces\IAjaxViewModel {

  public function EncodeToJson($vm) {
    header('Content-Type: application/json');
    return json_encode($vm, JSON_PRETTY_PRINT);
  }
}
